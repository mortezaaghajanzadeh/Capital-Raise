    /*Add Thousands Separator To Numbers */
    function addCommas(nStr) {
        nStr += '';
        x = nStr.split('.');
        x1 = x[0];
        x2 = x.length > 1 ? '.' + x[1] : '';
        var rgx = /(\d+)(\d{3})/;
        while (rgx.test(x1)) {
            x1 = x1.replace(rgx, '$1' + ',' + '$2');
        }
        return x1 + x2;
    }

        function menuheight() {
            if ($(window).width() > 767) {
                var maxHeight = -1;
                $(".menu-div").height("auto");
                var i = 0;
                $(".statement-content > div").each(function () {
                    i++;
                    maxHeight = maxHeight > $(this).height() ? maxHeight : $(this).height();
                })
                $(".menu-div").height(maxHeight);
            } else {
                $(".menu-div").height("auto");
            }
        }

            menuheight();
